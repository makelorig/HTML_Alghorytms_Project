    void BubbleSort(int[] bS)
    {
        for (int i = 0; i < bS.Length; i++)
        {
            for (int j = i; j < bS.Length; j++)
            {
                if (bS[j] < bS[i])
                {
                    int temp = bS[i];
                    bS[i] = bS[j];
                    bS[j] = temp;
                }
            }
        }
    }
